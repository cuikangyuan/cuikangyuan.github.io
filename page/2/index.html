数据结构学习笔记-11

循环链表合并

//假设A， B为非空循环链表的尾指针
LinkList Connect(LinkList A, LinkList B)
{
    //保存A表的头结点位置
    LinkList p = A->next;

    //B表的开始结点链接到A表尾
    A->next = B->next->next;

    //释放B表头结点
    free(B->next);

    B->next = p;

    //返回新循环链表的尾指针
    return B; 

}
